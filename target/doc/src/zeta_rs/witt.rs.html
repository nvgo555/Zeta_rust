1//! $p$-Adic Multi-Precision Lifting via Witt Vectors.
2//!
3//! Provides the `WittVector` structure which acts as a stack-allocated $p$-adic number 
4//! of length up to 4 elements. This is used for precise error accumulation across 
5//! learning steps.
6
7use crate::ring::{Z13Eta, P};
8use std::cmp::min;
9
10/// A multi-precision Witt Vector for $p$-adic arithmetic.
11///
12/// It uses a fixed-size `[Z13Eta; 4]` buffer to guarantee zero-allocation (stack-only) semantics
13/// while providing sufficient precision for tracking gradients/errors over time.
14#[derive(Clone, Copy, Debug, PartialEq, Eq)]
15pub struct WittVector {
16    /// The number of valid precision components (up to 4).
17    pub prec: usize,
18    /// The component data.
19    pub data: [Z13Eta; 4],
20}
21
22impl WittVector {
23    /// Lifts a basic ring element into a multi-precision Witt vector at position 0.
24    pub fn from_ring(a: Z13Eta, prec: usize) -> Self {
25        let mut data = [Z13Eta::ZERO; 4];
26        if prec > 0 {
27            data[0] = a;
28        }
29        WittVector { prec, data }
30    }
31
32    pub fn to_ring(&self) -> Z13Eta {
33        if self.prec > 0 {
34            self.data[0]
35        } else {
36            Z13Eta::ZERO
37        }
38    }
39
40    /// Adds two Witt Vectors component-wise with precise $p$-adic carry propagation.
41    pub fn wadd(x: &WittVector, y: &WittVector) -> WittVector {
42        let prec = min(min(x.prec, y.prec), 4);
43        let mut out = [Z13Eta::ZERO; 4];
44        let mut carry = [0u8; 3];
45        
46        for i in 0..prec {
47            let mut raw = [0u8; 3];
48            let mut next_carry = [0u8; 3];
49            for j in 0..3 {
50                let sum = x.data[i].0[j] as u16 + y.data[i].0[j] as u16 + carry[j] as u16;
51                raw[j] = (sum % P as u16) as u8;
52                next_carry[j] = (sum / P as u16) as u8;
53            }
54            out[i] = Z13Eta(raw);
55            carry = next_carry;
56        }
57        
58        WittVector { prec, data: out }
59    }
60
61    pub fn wneg(x: &WittVector) -> WittVector {
62        let mut out = [Z13Eta::ZERO; 4];
63        for i in 0..x.prec {
64            out[i] = -x.data[i];
65        }
66        WittVector { prec: x.prec, data: out }
67    }
68
69    pub fn wsub(x: &WittVector, y: &WittVector) -> WittVector {
70        Self::wadd(x, &Self::wneg(y))
71    }
72
73    /// Multiplies two Witt Vectors, expanding cross-terms and correctly tracking multi-level carries.
74    pub fn wmul(x: &WittVector, y: &WittVector) -> WittVector {
75        let prec = min(min(x.prec, y.prec), 4);
76        let mut l = [[0u16; 3]; 8]; 
77        
78        for i in 0..prec {
79            for j in 0..prec {
80                let a = x.data[i].0;
81                let b = y.data[j].0;
82                let c0 = a[0] as u16 * b[0] as u16;
83                let c1 = a[0] as u16 * b[1] as u16 + a[1] as u16 * b[0] as u16;
84                let c2 = a[0] as u16 * b[2] as u16 + a[1] as u16 * b[1] as u16 + a[2] as u16 * b[0] as u16;
85                let c3 = a[1] as u16 * b[2] as u16 + a[2] as u16 * b[1] as u16;
86                let c4 = a[2] as u16 * b[2] as u16;
87                
88                let exact = [
89                    c0 + c3 + c4,
90                    c1 + c3 + 2 * c4,
91                    c2 + c3 + 2 * c4,
92                ];
93                
94                let target_idx = i + j;
95                if target_idx < 2 * prec && target_idx < 8 {
96                    l[target_idx][0] += exact[0];
97                    l[target_idx][1] += exact[1];
98                    l[target_idx][2] += exact[2];
99                }
100            }
101        }
102        
103        let mut out = [Z13Eta::ZERO; 4];
104        let mut carry = [0u16; 3];
105        
106        for i in 0..prec {
107            let mut raw = [0u8; 3];
108            let mut next_carry = [0u16; 3];
109            for j in 0..3 {
110                let sum = l[i][j] + carry[j];
111                raw[j] = (sum % P as u16) as u8;
112                next_carry[j] = sum / P as u16;
113            }
114            out[i] = Z13Eta(raw);
115            carry = next_carry;
116        }
117        
118        WittVector { prec, data: out }
119    }
120}
