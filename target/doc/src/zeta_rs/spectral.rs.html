1//! Spectral Projectors and $T_3$ Dynamics.
2//!
3//! Implements the Sylvester projectors that separate the ring into its 1D and 2D Galois orbits,
4//! and the application of the unimodular matrix $T_3 \in \mathrm{SL}(3, \mathbb{Z})$.
5
6use crate::constants::{P1_MAT, P23_MAT, S3_MATS, S3_MUL, T3_POW};
7use crate::ring::{P, Z13Eta, ORD};
8
9pub fn proj(x: &Z13Eta, m: &[[u8; 3]; 3]) -> Z13Eta {
10    let x0 = x.0[0]; let x1 = x.0[1]; let x2 = x.0[2];
11    Z13Eta([
12        (x0 * m[0][0] + x1 * m[0][1] + x2 * m[0][2]) % P,
13        (x0 * m[1][0] + x1 * m[1][1] + x2 * m[1][2]) % P,
14        (x0 * m[2][0] + x1 * m[2][1] + x2 * m[2][2]) % P,
15    ])
16}
17
18/// Applies the $T_3$ transition matrix exactly $n$ times.
19///
20/// $T_3$ has characteristic polynomial $X^3 - X^2 - X - 1$ matching the ring extension.
21#[inline(always)]
22pub fn t3n(x: &Z13Eta, n: usize) -> Z13Eta {
23    proj(x, &T3_POW[n % ORD as usize])
24}
25
26pub fn t3n_s3(x: &Z13Eta, n: usize, g: usize) -> Z13Eta {
27    S3Galois::apply(&t3n(x, n), g)
28}

30pub struct SylvesterProjectors;
31
32impl SylvesterProjectors {
33    pub fn p1(x: &Z13Eta) -> Z13Eta {
34        proj(x, &P1_MAT)
35    }
36
37    pub fn p23(x: &Z13Eta) -> Z13Eta {
38        proj(x, &P23_MAT)
39    }
40
41    pub fn split(x: &Z13Eta) -> (Z13Eta, Z13Eta) {
42        let p1x = proj(x, &P1_MAT);
43        let p23x = *x - p1x;
44        (p1x, p23x)
45    }
46}
47
48pub struct SpectralDecomposition;
49
50impl SpectralDecomposition {
51    pub fn evolve(x: &Z13Eta, n: usize) -> Z13Eta {
52        let p1x = SylvesterProjectors::p1(x);
53        let mut p1_mult = 1;
54        for _ in 0..n { p1_mult = (p1_mult * 7) % P; }
55        
56        let term1 = p1x.smul(p1_mult);
57        let p23x = *x - p1x;
58        let term2 = t3n(&p23x, n);
59        
60        term1 + term2
61    }
62}
63
64pub struct S3Galois;
65
66impl S3Galois {
67    pub fn apply(x: &Z13Eta, g: usize) -> Z13Eta {
68        proj(x, &S3_MATS[g % 6])
69    }
70
71    pub fn compose_indices(g: usize, h: usize) -> usize {
72        S3_MUL[g % 6][h % 6]
73    }
74
75    pub fn orbit(x: &Z13Eta) -> [Z13Eta; 6] {
76        [
77            proj(x, &S3_MATS[0]),
78            proj(x, &S3_MATS[1]),
79            proj(x, &S3_MATS[2]),
80            proj(x, &S3_MATS[3]),
81            proj(x, &S3_MATS[4]),
82            proj(x, &S3_MATS[5]),
83        ]
84    }
85
86    pub fn casimir(x: &Z13Eta) -> Z13Eta {
87        let o = Self::orbit(x);
88        let mut sum = Z13Eta::ZERO;
89        for item in &o {
90            sum = sum + *item;
91        }
92        sum
93    }
