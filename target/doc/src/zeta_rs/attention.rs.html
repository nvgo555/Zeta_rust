1use ndarray::{Array3, Array2, concatenate, Axis};
2use crate::ring::Z13Eta;
3use crate::spectral::SylvesterProjectors;
4use crate::kernel::PAdicKernel;
5use crate::hilbert::HilbertEta;
6use crate::linear::{ring_lin, laplacian, w_seed, alg_dropout};
7
8pub struct SpectralAttention {
9    pub step: usize,
10    wq_dom: Array2<Z13Eta>,
11    wk_dom: Array2<Z13Eta>,
12    wv_dom: Array2<Z13Eta>,
13    wq_sub: Array2<Z13Eta>,
14    wk_sub: Array2<Z13Eta>,
15    wv_sub: Array2<Z13Eta>,
16    wo: Array2<Z13Eta>,
17}
18
19impl SpectralAttention {
20    pub fn new(d: usize) -> Self {
21        assert!(d % 6 == 0);
22        let hd = d / 3;
23        Self {
24            step: 0,
25            wq_dom: w_seed(d, hd),
26            wk_dom: w_seed(d, hd),
27            wv_dom: w_seed(d, hd),
28            wq_sub: w_seed(d, hd),
29            wk_sub: w_seed(d, hd),
30            wv_sub: w_seed(d, hd),
31            wo: w_seed(2 * hd, d),
32        }
33    }
34
35    pub fn forward(&self, x: &Array3<Z13Eta>, is_training: bool) -> Array3<Z13Eta> {
36        let shape = x.shape();
37        let (b, l, d) = (shape[0], shape[1], shape[2]);
38        let hd = d / 3;
39
40        let mut x_p1 = Array3::from_elem((b, l, d), Z13Eta::ZERO);
41        let mut x_p23 = Array3::from_elem((b, l, d), Z13Eta::ZERO);
42        
43        for i in 0..b {
44            for j in 0..l {
45                for k in 0..d {
46                    let (p1, p23) = SylvesterProjectors::split(&x[[i, j, k]]);
47                    x_p1[[i, j, k]] = p1;
48                    x_p23[[i, j, k]] = p23;
49                }
50            }
51        }
52        
53        let x_p1_k = PAdicKernel::apply(&x_p1);
54        let x_p23_k = PAdicKernel::apply(&x_p23);
55        
56        let q1 = ring_lin(&x_p1_k, &self.wq_dom);
57        let k1 = ring_lin(&x_p1_k, &self.wk_dom);
58        let v1 = ring_lin(&x_p1_k, &self.wv_dom);
59        
60        let q2 = ring_lin(&x_p23_k, &self.wq_sub);
61        let k2 = ring_lin(&x_p23_k, &self.wk_sub);
62        let v2 = ring_lin(&x_p23_k, &self.wv_sub);
63        
64        let mut o1 = Array3::from_elem((b, l, hd), Z13Eta::ZERO);
65        let mut o2 = Array3::from_elem((b, l, hd), Z13Eta::ZERO);
66        
67        for i in 0..b {
68            for j in 0..l {
69                for k in 0..hd {
70                    let s1 = (q1[[i, j, k]] * k1[[i, j, k]]).trace();
71                    let s2 = (q2[[i, j, k]] * k2[[i, j, k]]).trace();
72                    
73                    o1[[i, j, k]] = v1[[i, j, k]].smul(s1);
74                    o2[[i, j, k]] = v2[[i, j, k]].smul(s2);
75                }
76            }
77        }
78        
79        let combined = concatenate(Axis(2), &[o1.view(), o2.view()]).unwrap();
80        let out_proj = ring_lin(&combined, &self.wo);
81        
82        let mut out = Array3::from_elem((b, l, d), Z13Eta::ZERO);
83        for i in 0..b {
84            for j in 0..l {
85                for k in 0..d {
86                    out[[i, j, k]] = x[[i, j, k]] + out_proj[[i, j, k]];
87                }
88            }
89        }
90        
91        let mut out = laplacian(&out);
92        HilbertEta::layer_norm(&mut out);
93        
94        alg_dropout(&mut out, 3, self.step, is_training); // DROP = 3
95        
96        out
97    }
98}
